// 我的工具类
public class MyTool
{
    public static void main(String[] args)
    {
        int a = 100;
        int b = 20;
        if(a>b && true) // a 大于 b 并且 true
            System.out.println(a);
        else
            System.out.println("this! aaakkk");  // 测试注释显示是否正确
    }
}
